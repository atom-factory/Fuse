129  void Direct2DBackend::DrawPath(const Offset& position,
130  Vector<Offset> points,
131  const Color& fillColor,
132  const Stroke& stroke) {
133  if (points.size() < 3) {
134  throw std::runtime_error("Direct2DBackend::DrawPath: Too few points.");
135  }
136 
137  if (m_RenderTarget && m_Factory) {
138  ID2D1PathGeometry* pathGeometry = nullptr;
139 
140  auto hr = m_Factory->CreatePathGeometry(&pathGeometry);
141  CheckResult(hr, "Failed to create path geometry.");
142 
143  // Generate path geometry from provided points
144  {
145  ID2D1GeometrySink* sink = nullptr;
146  hr = pathGeometry->Open(&sink);
147  CheckResult(hr, "Failed to open path geometry.");
148 
149  const auto d2dPoints = PrivateOffsetVecToD2DPointVec(points);
150  sink->BeginFigure(d2dPoints.front(), D2D1_FIGURE_BEGIN_FILLED);
151 
152  // TODO: Calculate the center point of the shape and make that the origin point for
153  // applying transformations.
154  // Each point X gets transformed by taking Cx + X, same for Y.
155  const auto center = Polygon::CalculateCenter(points);
156 
157  for (size_t i = 1; i < points.size(); ++i) {
158  const auto x = position.X + d2dPoints.at(i).x;
159  const auto y = position.Y + d2dPoints.at(i).y;
160 
161  sink->AddLine(D2D1::Point2F(x, y));
162  }
163 
164  sink->EndFigure(D2D1_FIGURE_END_CLOSED);
165  hr = sink->Close();
166  CheckResult(hr, "Failed to close path geometry.");
167  sink->Release();
168  }
169 
170  // Draw the generated geometry
171  {
172  ID2D1SolidColorBrush* fillBrush = nullptr;
173  hr = m_RenderTarget->CreateSolidColorBrush(D2D1::ColorF(fillColor.Value()),
174  &fillBrush);
175  CheckResult(hr, "Failed to create D2D brush.");
176 
177  m_RenderTarget->FillGeometry(pathGeometry, fillBrush);
178  fillBrush->Release();
179 
180  if (stroke.Thickness > 0) {
181  ID2D1SolidColorBrush* strokeBrush = nullptr;
182  hr = m_RenderTarget->CreateSolidColorBrush(D2D1::ColorF(stroke.Color.Value()),
183  &strokeBrush);
184  CheckResult(hr, "Failed to create D2D brush.");
185  m_RenderTarget->DrawGeometry(pathGeometry,
186  strokeBrush,
187  stroke.Thickness,
188  nullptr);
189  strokeBrush->Release();
