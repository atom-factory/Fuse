24  void IPluginView::OnPaint() {
25  if (const auto backend = GetBackend()) {
26  if (m_OwningCanvas) {
27  // Clear drawing area (canvas) with our specified background color
28  backend->BeginDrawing(m_OwningCanvas->BackgroundColor);
29 
30  // Draw our component tree starting with the root returned from
31  // IPluginCanvas::Draw() using a breadth-first search
32  {
33  const auto dimension = Dimension(m_ViewState.Size);
34  const auto root = m_OwningCanvas->Draw(this, dimension);
35  std::queue<IComponent*> queue;
36  queue.push(root);
37 
38  while (!queue.empty()) {
39  IComponent* component = queue.front();
40  queue.pop();
41 
42  component->Draw(backend);
43 
44  auto result = component->GetChildren();
45  if (result.has_value()) {
46  auto children = result.value();
47  for (const auto& child : children) {
48  queue.push(child);
49  }
50  }
51  }
52  }
53 
54  backend->EndDrawing();
55  }
56  }
