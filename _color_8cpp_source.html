85  Color Color::AlphaBlend(const Color& foreground, const Color& background) {
86  const f32 alpha = foreground.Alpha();
87  if (alpha == 0.f) {
88  return background;
89  }
90  const f32 invAlpha = 1.f - alpha;
91 
92  f32 backAlpha = background.Alpha();
93  if (backAlpha == 1.f) {
94  return {alpha * foreground.Red() + invAlpha * background.Red(),
95  alpha * foreground.Green() + invAlpha * background.Green(),
96  alpha * foreground.Blue() + invAlpha * background.Blue(),
97  1.f};
98  }
99 
100  backAlpha = (backAlpha * invAlpha) / 1.f;
101  const f32 outAlpha = alpha + backAlpha;
102  assert(outAlpha != 0.f);
103  return {alpha * foreground.Red() + invAlpha * background.Red() / outAlpha,
104  alpha * foreground.Green() + invAlpha * background.Green() / outAlpha,
105  alpha * foreground.Blue() + invAlpha * background.Blue() / outAlpha,
106  1.f};
107  }
