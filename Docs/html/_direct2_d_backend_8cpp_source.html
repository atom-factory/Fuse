12 void Direct2DBackend::Init() {
13 if (!m_OwningView) {
14 throw std::runtime_error(
15 "Direct2DBackend::Init: m_OwningView is null. Ensure SetOwner() "
16 "was called before initializing.");
17 }
18
19 auto hr = ::D2D1CreateFactory(D2D1_FACTORY_TYPE_SINGLE_THREADED, &m_Factory);
20 if (FAILED(hr)) {
21 throw std::runtime_error("Failed to create Direct2D Factory");
22 }
23
24 auto view = m_OwningView->As<Win32PluginView>();
25 RECT rc;
26 ::GetClientRect(view->GetHandle(), &rc);
27 hr = m_Factory->CreateHwndRenderTarget(
28 D2D1::RenderTargetProperties(),
29 D2D1::HwndRenderTargetProperties(view->GetHandle(),
30 D2D1::SizeU(rc.right - rc.left, rc.bottom - rc.top)),
31 &m_RenderTarget);
32
33 if (FAILED(hr)) {
34 throw std::runtime_error("Failed to create render target");
35 }
36 }
